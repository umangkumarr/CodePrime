#include <iostream>
using namespace std;

int ModularExponent(int a, int n, int mod) {
    int ans = 1;
    while (n > 0) {
        int bit = n & 1;
        if (bit) {
            ans = (ans * a) % mod;
        }
        n >>= 1;
        a = (a * a) % mod;
    }
    return ans;
}
