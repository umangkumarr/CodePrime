void solve() {
    cin >> n >> k;
    vl a(n);
    for (int i = 0; i < n; i++) cin >> a[i];

    vl ans(n, INT_MAX);
    ans[0] = 0;
    for (int i = 1; i < n; i++) {
        for (int j = i - 1; j >= 0 && j >= i - k; j--) {
            ans[i] = min(ans[i], abs(a[i] - a[j]) + ans[j]);
        }
    }
    cout << ans[n - 1];
}
